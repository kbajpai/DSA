2  { 
 
3    for (var i = 1; i < a.Length; i++) 
 
4    { 
 
5      var n = a[i]; 
 
6      var j = i; 
 
7      while (j > 0 && n < a[j - 1]) 
 
8      { 
 
9        a[j] = a[j - 1]; 
 
10        j--; 
 
11      } 
 
12 
 
13      a[j] = n; 
 
14    } 
 
15 
 
16    return a; 
 
17  }
