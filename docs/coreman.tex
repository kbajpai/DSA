	\begin{lstlisting}[language={[Sharp]C}, caption={Insertion Sort}, label={Script}]
	public static int[] InsertionSort(int[] a)
	{
		for (var i = 1; i < a.Length; i++)
		{
			var n = a[i];
			var j = i;
			while (j > 0 && n < a[j - 1])
			{
				a[j] = a[j - 1];
				j--;
			}

			a[j] = n;
		}

		return a;
	}
	\end{lstlisting}
